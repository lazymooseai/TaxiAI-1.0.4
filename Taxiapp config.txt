def _require(key: str) -> str:
“”“Pakollinen ympäristömuuttuja — kaatuu selkeällä viestillä.”””
value = os.environ.get(key)
if not value:
raise EnvironmentError(
f”[Config] Pakollinen ympäristömuuttuja puuttuu: {key}\n”
f”Aseta Streamlit Cloud → Secrets tai .streamlit/secrets.toml”
)
return value

# VANHA (BUGI): time.daylight — vakio, ei kerro onko kesäaika NYT.

# UUSI: zoneinfo laskee oikean offsetin aina automaattisesti.

HELSINKI_TZ = ZoneInfo(“Europe/Helsinki”)

def helsinki_now() -> datetime:
“”“Palauta Helsingin paikallisaika (timezone-aware).”””
return datetime.now(HELSINKI_TZ)

def helsinki_offset() -> int:
“”“Palauta Helsingin UTC-offset (2 talvella, 3 kesällä).”””
now = datetime.now(timezone.utc)
return int(now.astimezone(HELSINKI_TZ).utcoffset().total_seconds() / 3600)

def to_helsinki(dt: datetime) -> datetime:
“”“Muunna mikä tahansa datetime Helsinki-aikaan.”””
if dt.tzinfo is None:
dt = dt.replace(tzinfo=timezone.utc)
return dt.astimezone(HELSINKI_TZ)
